let selectedIndex=0; let selectedGroup=0; function focusItem(index){ const items= document.querySelectorAll(".item"); if(!items.length) return; items.forEach(i=> i.classList.remove("tv-focus") ); if(index<0) index=items.length-1; if(index>=items.length) index=0; selectedIndex=index; items[index] .classList .add("tv-focus"); items[index].scrollIntoView({ behavior:"smooth", block:"nearest" }); } function changeGroup(dir){ const groups= document.querySelectorAll( ".group-btn" ); if(!groups.length) return; selectedGroup+=dir; if(selectedGroup<0) selectedGroup= groups.length-1; if( selectedGroup>= groups.length ){ selectedGroup=0; } groups[selectedGroup].click(); setTimeout(()=>{ focusItem(0); },200); } document.addEventListener( "keydown", function(e){ switch(e.keyCode){ case 38: focusItem( selectedIndex-1 ); break; case 40: focusItem( selectedIndex+1 ); break; case 37: changeGroup(-1); break; case 39: changeGroup(1); break; case 13: const items= document.querySelectorAll( ".item" ); if( items[selectedIndex] ){ items[ selectedIndex ].click(); } break; case 166: focusItem( selectedIndex+1 ); const up= document.querySelectorAll( ".item" ); if(up[selectedIndex]) up[selectedIndex] .click(); break; case 167: focusItem( selectedIndex-1 ); const down= document.querySelectorAll( ".item" ); if(down[selectedIndex]) down[selectedIndex] .click(); break; } });
